% 定义半加器模块
\newcommand{\halfadder}{
    % 绘制半加器矩形框
    \draw (0,0) rectangle (3,2);

    % 在框内添加"半加器"文字
    \node at (1.5, 1) {半加器};

    % 输入端口A，B，输出端口S，CO
    \draw (0,1.5) node[right] {A} -- ++(-0.5,0); % 输入A
    \draw (0,0.5) node[right] {B} -- ++(-0.5,0); % 输入B
    \draw (3,1.5) node[left] {S} -- ++(0.5,0);  % 输出S
    \draw (3,0.5) node[left] {CO} -- ++(0.5,0); % 输出CO
}
